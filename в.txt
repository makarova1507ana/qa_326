

CREATE TABLE products(
    id INT AUTO_INCREMENT PRIMARY KEY, 
    nameP VARCHAR(50),
    type VARCHAR(50),
    cost FLOAT DEFAULT 0); /*DEFAULT -  по умолчанию*/
    
CREATE TABLE customers(
    id INT AUTO_INCREMENT PRIMARY KEY, 
    nameC VARCHAR(50),
    adress VARCHAR(50)
);
CREATE TABLE customer_product(
    id_customer INT,
    id_product INT,
    FOREIGN KEY(id_customer) REFERENCES customers(id)
      
);
